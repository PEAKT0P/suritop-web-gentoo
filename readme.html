Установка на чистый Gentoo + OpenRC

1. Установить sudo 
root@server # emerge --sync
root@server # emerge -av app-admin/sudo
root@server # echo "pro ALL=(ALL) NOPASSWD: ALL" > /etc/sudoers.d/pro
2. Скачать и распаковать overlay 
root@server # cd /tmp && tar xzf suritop-web-overlay.tar.gz
3. Зарегистрировать overlay 
root@server # mkdir -p /etc/portage/repos.conf
root@server # cat > /etc/portage/repos.conf/suritop-web.conf
[suritop-web]
location = /usr/local/portage/local
masters = gentoo
auto-sync = false
root@server # mkdir -p /usr/local/portage/local
root@server # cp -r /tmp/suritop-web-overlay/* /usr/local/portage/local/
4. Принять ключи 
root@server # mkdir -p /etc/portage/package.accept_keywords
root@server # cat > /etc/portage/package.accept_keywords/suritop-web
net-analyzer/suritop-web **
net-analyzer/suricata **
net-libs/libhtp **
5. Установить пакет 
root@server # emerge -av --backtrack=30 net-analyzer/suritop-web
6. Интерактивная настройка 
root@server # emerge --config net-analyzer/suritop-web

USE Flags

FlagDefaultОписание
+iptablesYESАвтогенерация iptables правил
-natNONAT masquerade (раздача интернета)
-dockerNODocker интеграция (DOCKER-USER chain)
+suricataYESIPS правила Suricata
-geoipNOGeoIP определение

Как использовать USE flags

# Стандартная установка (без автогенерации файрвола)
root@server # cat > /etc/portage/package.use/suritop-web
net-analyzer/suritop-web -iptables -nat -docker +suricata

# С автогенерацией файрвола
net-analyzer/suritop-web +iptables +nat +docker +suricata

# Только мониторинг (без файрвола)
net-analyzer/suritop-web -iptables -nat -docker -suricata

Конфигурация

/etc/conf.d/suritop-web

# Автоматическая настройка iptables при загрузке
# ВНИМАНИЕ: заменяет все существующие правила!
SURITOP_AUTO_IPTABLES="no"

# Сеть
SURITOP_SERVER_IP="172.23.230.169"
SURITOP_NET_INTERFACE="enp6s0"
SURITOP_SSH_PORT="61122"

# Whitelist IP (никогда не блокировать)
SURITOP_WHITELIST_IP="172.23.230.169 172.23.224.1"

# БД
SURITOP_DB_NAME="server_stats"
SURITOP_DB_USER_R="stats_reader"
SURITOP_DB_PASS_R="suritop_read_2026"
SURITOP_DB_USER_W="stats_writer"
SURITOP_DB_PASS_W="suritop_write_2026"

# NAT (для раздачи интернета)
SURITOP_NAT_ENABLE="no"

# Docker
SURITOP_DOCKER_ENABLE="no"
SURITOP_DOCKER_BRIDGE="172.17.0.0/16"
